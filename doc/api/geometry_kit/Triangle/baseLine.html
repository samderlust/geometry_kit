/// Get baseLine of this triangle
Line get baseLine {
  Point start, end;
  if (a.x <= b.x && a.x <= c.x) {
    start = a;
    end = b.x <= c.x ? b : c;
  } else if (b.x <= a.x && b.x <= c.x) {
    start = b;
    end = a.x <= c.x ? a : c;
  } else {
    start = c;
    end = a.x <= b.x ? a : b;
  }
  return Line(start, end);
}
